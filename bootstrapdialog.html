The render example is initialized with the following code 

rndDialog.addButton({
  key: 'cancel',
  caption: 'Cancel'
}, false, true, 'pull-left');
    
rndDialog.addButton(
  dom.createDom('button', {
      'class': 'btn btn-primary pull-right',
      'name': 'lovely'
    }, dom.createTextNode('Love it '),
    dom.createDom('span', {
      'class': 'glyphicon glyphicon-heart'
    })),
  true);

rndDialog.setTitle('Render demo');
rndDialog.setBodyContent(dom.createDom('p',
  null, 'This is a dynamically created dialog'));
